vmalloc的数据组织

2017-02-25

1. vm_struct和vmap_area

vmalloc分配出来的区间 用 vm_struct 表示.

linux-3.10.86/include/linux/vmalloc.h

struct vm_struct {
    struct vm_struct    *next;
    ...
};

vm_struct 是通过单向链表把vmalloc区域串起来的, 但要找寻一个区域的话, 单向链表的效率就太低了, 所以, 有struct vmap_area, 且用红黑树vmap_area_root来管理.

struct vm_struct * __get_vm_area_node(...)
{
    alloc_vmap_area
    {
        搜索 vmap_area_root
    }
}

我们找到一个区间vmap_area, 并有个对应的struct vm_struct:

__get_vm_area_node
{

    struct vm_struct *area=kzalloc_node();
    struct vmap_area *va = alloc_vmap_area();
    setup_vmalloc_vm(struct vm_struct *vm, struct vmap_area *va, ...)
    {
        vm->addr = (void *)va->va_start;
        vm->size = va->va_end - va->va_start;
        va->vm = vm;
        va->flags |= VM_VM_AREA;
    }
}

我们只通过va来得到vm, 不从vm得到va. (纲领是va)

问题: struct vm_struct 和struct vmap_area 两者有重复的信息, 为何不考虑 两个合成一个结构体呢?
答: 可能是生命周期不一样的:
1. 对 struct vmap_area 的释放 采取的lazy的方式,
2. struct vm_struct 却是立刻释放.

2. 移除

看下移除的时候, 如何处理vmap_area和vm_struct:

vfree(const void *addr)
|--__vunmap(const void *addr, ...)
|   |--struct vm_struct *area = remove_vm_area
|   |--free 该区间的page frame
|   |--kfree(area)


remove_vm_area
{
    struct vmap_area *va= find_vmap_area(...);
    spin_lock(&vmap_area_lock);
    /*并没有从红黑树上移除哦, 
    只是和 struct vmap_area 和 struct vm_struct 断了联系.
    struct vm_struct内无field来得知对应的vmap_area的.
    */
    va->vm = NULL;
    va->flags &= ~VM_VM_AREA;
    spin_unlock(&vmap_area_lock);

    free_unmap_vmap_area(va);
}

free_unmap_vmap_area
|--free_unmap_vmap_area_noflush
|   |--unmap_vmap_area //页表相关
|   |--free_vmap_area_noflush


static void free_vmap_area_noflush(struct vmap_area *va)
{
    va->flags |= VM_LAZY_FREE;  //通过此标志来得知, 该区域未 purge, 待集中处理.
    atomic_add((va->va_end - va->va_start) >> PAGE_SHIFT, &vmap_lazy_nr);
    if (unlikely(atomic_read(&vmap_lazy_nr) > lazy_max_pages()))
        try_purge_vmap_area_lazy();
}

try_purge_vmap_area_lazy -> __purge_vmap_area_lazy
|--__free_vmap_area
|   |-- list_del_rcu(&va->list);

4.struct vm_area_struct 和 vm_struct的区别

vm_area_struct是进程的虚拟地址区间.
vm_struct 用来表示 vmalloc分配出来的区间.
两者在区间的数值上也就存在差别, 一个0~3g, 另一个VMALLOC_START ~ VMALLOC_END.
